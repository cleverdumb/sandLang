atom Empty alias E {
    section property {
        cdef render 0
    }
}

atom Sand alias S {
    section property {
        cdef render 1
        cdef color #FF0000
        cdef key s
        cdef size 10

        cdef dragCD 1
    }
    section definition {
        def R <^E>
    }
    section update {
        ext sandLike <repl=R, prob = 1>
    }
}
