atom Empty alias E {
    section property {
        cdef render 0
    }
}

atom Sand alias S {
    section property {
        cdef render 1
        cdef color #FF0000
        cdef key s
        cdef size 10

        cdef dragCD 0
    }
    section definition {
        def R <^E>
    }
    section update {
        match (0, 0, 2, 2) {
            pattern
            x *
            * _
        }
        -> {
            pattern
            _ /
            / x
            shift (1, 1)
            always-run
        }
    }
}
